    subroutine compute_indices(me)

    implicit none

    class(sparsity_pattern),intent(inout) :: me

    integer :: i !! counter

    allocate(me%indices(me%num_nonzero_elements))
    !me%indices = [(i,i=1,me%num_nonzero_elements)]

        me%indices(i) = i
    end do
